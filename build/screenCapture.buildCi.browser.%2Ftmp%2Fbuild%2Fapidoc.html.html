function time-require.requireHook (listener, autohook)

  • description and source-code
    requireHook = function (listener, autohook) {
	if (typeof listener !== "function") {
		throw new Error("The hooking function should be set");
	}
	// set the listener
	_listener = listener;
	// if autohook (by default),
	if (autohook !== false) {
		_hook();
	}
	return {
		hookedAt: _hookedAt,
		hook: _hook,
		unhook: _unhook
	};
}
